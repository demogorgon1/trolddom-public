!define local MERCENARY_CONTRACT_ITEM
{
	type: miscellaneous
	item_level: @_level
	required_level: @_level
	rarity: uncommon
	icon: @_icon
	stack: 10
	string: @_item_name
	use_ability: @_ability
	cost: @_cost
}

!define local MERCENARY_CONTRACT_ABILITY
{
	string: @_item_name
	description: "Summons a mercenary. Lasts for 15 minutes."
	flags: [ target_self item minion_summon ]
	cast_time: 20
	icon: icon_scroll_1
	required_level: @_level
	cooldowns: [ global ]	
	consume_items: @_ability_consume

	direct_effect spawn_entity:
	{
		requirement self<must_not_have_aura>: emblem_of_triumph_trinket
	
		spawn_flags: [ no_owner ]
		entity: @_entity
		init_state: spawning
	}
	
	direct_effect spawn_entity:
	{
		requirement self<must_have_aura>: emblem_of_triumph_trinket
	
		spawn_flags: [ no_owner source_level ]
		entity: @_entity_level_match
		init_state: spawning
				
		refresh_npc_metrics: 
		{
			weapon_damage: @_weapon_damage
			health: @_resource_health
			armor: @_armor
			
			update: true # Refresh stats even if already defined
		}
	}	
}
